Create database ineuron;

  `id` INT NOT NULL AUTO_INCREMENT,
  `username` VARCHAR(45) NOT NULL UNIQUE,
  `lastname` VARCHAR(45) NULL,
  `firstname` VARCHAR(45) NULL,
`password` VARCHAR(45) NOT NULL,  `role` VARCHAR(45) NOT NULL,
  PRIMARY KEY (`id`));
